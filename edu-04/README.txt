# Build application release:
	1. mvn clean
	2. mvn install => $USER_HOME\.m2\repository\edu-04\gui-app\master\gui-app-master-release.zip

# Install and run application:
	1. Unzip gui-app-master-release.zip
	2. cd gui-app-master
	3. Double click: 'gui-app-master.jar' or 'run.bat'
